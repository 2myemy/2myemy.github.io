#include <cstdio>
using namespace std;

long mat[93];
int main() {
	int i, n;
	scanf("%d", &n);
	mat[0] = -1;
	mat[1] = 1;
	
	for (i=0; i<=n; i++) {
		mat[i+2] = mat[i] + mat[i+1];
	}
	
	printf("%ld", mat[n+2]);
	
	return 0;
}
