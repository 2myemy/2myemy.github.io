#include <cstdio>
using namespace std;

const int MAX = 100;
int matA[MAX][MAX];
int matB[MAX][MAX];
int matC[MAX][MAX];
void calc(int n, int m, int k);
int main() {
	int n, m, k;
	int i, j;
	scanf("%d %d", &n, &m);
	
	for (i=0; i<n; i++) {
		for (j=0; j<m; j++) {
			scanf("%d", &matA[i][j]);
		}
	}
	
	scanf("%d %d", &m, &k);
	
	for (i = 0; i < m; i++) {
		for (j =0; j<k;j++) {
			scanf("%d", &matB[i][j]);
		}
	}
	
	calc(n, m, k);
	
	for(i=0; i<n; i++){
		for(j=0;j<k;j++){
			printf("%d ", matC[i][j]);
		}
		printf("\n");
	}
	
	return 0;
}

void calc(int n, int m, int k) {
	
	int i, j, l;
	
	for (i=0; i<n; i++) {
		for (j =0; j < k; j++) {
			matC[i][j]=0;
		}
	}
	
	for (i =0; i < n; i++) {
		for (j=0;j<m;j++) {
			for (l=0; l<k;l++) {
				matC[i][l] += matA[i][j] * matB[j][l];
			}
		}
	}
}
