#include <cstdio>
using namespace std;

int calc(int n) {
	int sum = n;
	while(n != 0) {
		sum += (n % 10);
		n /= 10;
	}
	return sum;
}

int main() {
	int n, i;
	scanf("%d", &n);
	
	for (i = 0; i < 1000000; i++) {
		if (calc(i) == n) {
			printf("%d", i);
			return 0;
		}
	}
	printf("0");
	return 0;
}
